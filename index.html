Tratamento de Dados com Python

Este código em Python carrega uma planilha Excel contendo dados de vendas, realiza o tratamento da coluna "Preço", remove o símbolo "R$", converte os valores para numéricos e faz o agrupamento por "Vendedor" e "Produto". Por fim, o código salva os dados tratados em uma nova planilha Excel.

Código Python
import pandas as pd

# Carregar a planilha 
file_path = r'C:\Users\marco\Python\vendaDeProduto.xlsx'
df = pd.read_excel(file_path)

# Exibir as primeiras linhas para verificar o conteúdo
print("Dados iniciais:")
print(df.head())

# Tratar a coluna 'Preço' para remover o 'R$' e converter para numérico
df['Preço'] = df['Preço'].replace({r'R\$ ': '', ',': '.'}, regex=True).astype(float)

# Exibir as primeiras linhas após o tratamento do preço
print("\nDados após o tratamento da coluna 'Preço':")
print(df.head())

# Agrupar os dados por 'Vendedor' e calcular a soma das vendas
vendas_por_vendedor = df.groupby('Vendedor')['Preço'].sum().reset_index()

# Exibir o total de vendas por vendedor
print("\nTotal de vendas por vendedor:")
print(vendas_por_vendedor)

# Agrupar os dados por 'Produto' e calcular a soma das vendas
vendas_por_produto = df.groupby('Produto')['Preço'].sum().reset_index()

# Exibir o total de vendas por produto
print("\nTotal de vendas por produto:")
print(vendas_por_produto)

# Salvar o DataFrame tratado em um novo arquivo Excel (se necessário)
df.to_excel('venda_de_produto_tratada.xlsx', index=False)

print("\nArquivo tratado salvo como 'venda_de_produto_tratada.xlsx'")

Aplicação de Envio Automático de PDF e Mensagens no WhatsApp

Esta aplicação automatiza o envio de mensagens e arquivos PDF para números de telefone armazenados em um arquivo Excel, utilizando a biblioteca Puppeteer para interagir com o WhatsApp Web. Ela lê números de telefone de um arquivo Excel, faz login no WhatsApp Web, e envia o PDF especificado para os contatos de forma automatizada, incluindo controle de erro e captura de logs.

Código da Aplicação
const puppeteer = require('puppeteer');
const XLSX = require('xlsx');
const path = require('path');
const fs = require('fs');

// Lê números de telefone do arquivo Excel
function lerNumerosExcel(nomeArquivo) {
    const workbook = XLSX.readFile(nomeArquivo);
    const sheetName = workbook.SheetNames[0];
    const sheet = workbook.Sheets[sheetName];
    const dados = XLSX.utils.sheet_to_json(sheet);
    return dados.map(row => row.telefone);
}

// Caminho do arquivo PDF a ser enviado
const pdfPath = path.join(__dirname, '2025_ONE PAGE_MG3097NR.pdf');

// Caminho do arquivo Excel
const excelPath = path.join(__dirname, 'contato.xlsx');

(async () => {
    if (!fs.existsSync(excelPath)) {
        console.log(`Erro: O arquivo ${excelPath} não foi encontrado.`);
        return;
    }

    const browser = await puppeteer.launch({ headless: false });
    const page = await browser.newPage();
    await page.goto('https://web.whatsapp.com');

    console.log('Escaneie o QR Code para continuar...');

    try {
        await page.waitForSelector('canvas[aria-label="Scan me!"]', { timeout: 60000 });
        await page.waitForFunction(() => !document.querySelector('canvas[aria-label="Scan me!"]'), { timeout: 60000 });
    } catch (error) {
        console.log('QR Code não detectado ou já logado. Prosseguindo...');
    }

    try {
        await page.waitForSelector('div[contenteditable="true"][data-tab="3"]', { timeout: 60000 });
    } catch (error) {
        console.log('Erro ao carregar WhatsApp Web. Verifique a conexão.');
        await browser.close();
        return;
    }

    console.log('Login realizado com sucesso!');

    const numeros = lerNumerosExcel(excelPath);

    for (const numero of numeros) {
        console.log(`Enviando para: ${numero}`);
        await page.goto(`https://web.whatsapp.com/send?phone=${numero}&text=`);

        try {
            await page.waitForSelector('div[contenteditable="true"]', { timeout: 30000 });
        } catch (error) {
            console.log(`Não foi possível abrir o chat com ${numero}. Pulando para o próximo...`);
            continue;
        }

        await new Promise(resolve => setTimeout(resolve, 5000));

        try {
            await page.waitForSelector('span[data-icon="plus"]', { timeout: 30000, visible: true });
            await page.click('span[data-icon="plus"]');
            await new Promise(resolve => setTimeout(resolve, 1000));
        } catch (error) {
            console.log(`Falha ao encontrar o botão '+' para ${numero}. Pulando...`);
            continue;
        }

        try {
            await page.waitForSelector('span.xdod15v', { timeout: 30000, visible: true });
            await page.click('span.xdod15v');
        } catch (error) {
            console.log(`Falha ao encontrar o botão 'Documentos' para ${numero}. Pulando...`);
            continue;
        }

        try {
            await new Promise(resolve => setTimeout(resolve, 2000));
            const inputFile = await page.$('input[type="file"]');
            if (!inputFile) {
                console.log(`Seletores de arquivos não encontrados para ${numero}. Pulando...`);
                return;
            }

            await inputFile.uploadFile(pdfPath);
            console.log(`Arquivo PDF selecionado para ${numero}.`);
        } catch (error) {
            console.log(`Erro ao anexar arquivo para ${numero}. Pulando...`);
            await page.screenshot({ path: `erro_anexar_arquivo_${numero}.png` });
            continue;
        }

        try {
            await page.waitForSelector('span[data-icon="send"]', { timeout: 10000 });
            await page.click('span[data-icon="send"]');
            console.log(`Arquivo enviado para ${numero}`);
        } catch (error) {
            console.log(`Falha ao enviar mensagem para ${numero}. Pulando...`);
            continue;
        }

        await new Promise(resolve => setTimeout(resolve, 5000));
    }

    console.log('Envios finalizados!');
    await browser.close();
})();
